739  PolyType ElementNullSchemeMultiply(const PolyType &c1, const PolyType &c2,
740  const BigInteger &ptmod) const {
741  PolyType cResult(c1.GetParams(), Format::COEFFICIENT, true);
742 
743  if (!c1.GetParams()->OrderIsPowerOfTwo()) {
744  PALISADE_THROW(
745  not_implemented_error,
746  "Polynomial multiplication in coefficient representation is not "
747  "currently supported for non-power-of-two polynomials");
748  }
749 
750  PolyType cLarger(c1.GetParams(), Format::COEFFICIENT, true);
751 
752  IntType ptm(ptmod.ConvertToInt());
753 
754  int ringdim = c1.GetRingDimension();
755  for (int c1e = 0; c1e < ringdim; c1e++) {
756  IntType answer, c1val, c2val, prod;
757  c1val = c1.at(c1e);
758  if (c1val != IntType(0)) {
759  for (int c2e = 0; c2e < ringdim; c2e++) {
760  c2val = c2.at(c2e);
761  if (c2val != IntType(0)) {
762  prod = c1val * c2val;
763 
764  int index = (c1e + c2e);
765  if (index >= ringdim) {
766  index %= ringdim;
767  cLarger.at(index) = (cLarger.at(index) + prod) % ptm;
768  } else {
769  cResult.at(index) = (cResult.at(index) + prod) % ptm;
770  }
771  }
772  }
773  }
774  }
775 
776  // fold cLarger back into the answer
777  for (int i = 0; i < ringdim; i++) {
778  IntType adj = cResult.at(i) + (ptm - cLarger.at(i)) % ptm;
779  cResult.at(i) = adj % ptm;
780  }
781 
782  return cResult;
783  }
