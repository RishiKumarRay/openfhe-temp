443  bool operator==(const CiphertextImpl<Element>& rhs) const {
444  if (!CryptoObject<Element>::operator==(rhs)) return false;
445 
446  if (this->m_depth != rhs.m_depth) return false;
447 
448  if (this->m_level != rhs.m_level) return false;
449 
450  if (this->m_hopslevel != rhs.m_hopslevel) return false;
451 
452  if (this->m_scalingFactor != rhs.m_scalingFactor) return false;
453 
454  if (this->m_scalingFactorInt != rhs.m_scalingFactorInt) return false;
455 
456  const std::vector<Element>& lhsE = this->GetElements();
457  const std::vector<Element>& rhsE = rhs.GetElements();
458 
459  if (lhsE.size() != rhsE.size()) return false;
460 
461  for (size_t i = 0; i < lhsE.size(); i++) {
462  const Element& lE = lhsE[i];
463  const Element& rE = rhsE[i];
464 
465  if (lE != rE) return false;
466  }
467 
468  const shared_ptr<std::map<string, shared_ptr<Metadata>>> lhsMap =
469  this->m_metadataMap;
470  const shared_ptr<std::map<string, shared_ptr<Metadata>>> rhsMap =
471  rhs.m_metadataMap;
472 
473  if (lhsMap->size() != rhsMap->size()) return false;
474 
475  if (lhsMap->size() > 0) {
476  for (auto i = lhsMap->begin(), j = rhsMap->begin(); i != lhsMap->end();
477  ++i, ++j)
478  if (!(*(i->second) == *(j->second))) return false;
479  }
480 
481  return true;
482  }
