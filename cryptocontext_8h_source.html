891  const std::vector<std::complex<double>>& value, size_t depth = 1,
892  uint32_t level = 0, const shared_ptr<ParmType> params = nullptr) const {
893  Plaintext p;
894  const auto cryptoParams =
895  std::dynamic_pointer_cast<CryptoParametersRNS>(
896  GetCryptoParameters());
897 
898  double scFact = cryptoParams->GetScalingFactorReal(level);
899 
900  if (params == nullptr) {
901  shared_ptr<ILDCRTParams<DCRTPoly::Integer>> elemParamsPtr;
902  if (level != 0) {
903  ILDCRTParams<DCRTPoly::Integer> elemParams =
904  *(cryptoParams->GetElementParams());
905  for (uint32_t i = 0; i < level; i++) {
906  elemParams.PopLastParam();
907  }
908  elemParamsPtr =
909  std::make_shared<ILDCRTParams<DCRTPoly::Integer>>(elemParams);
910  } else {
911  elemParamsPtr = cryptoParams->GetElementParams();
912  }
913 
914  p = Plaintext(std::make_shared<CKKSPackedEncoding>(
915  elemParamsPtr, this->GetEncodingParams(), value, depth, level,
916  scFact));
917  } else {
918  p = Plaintext(std::make_shared<CKKSPackedEncoding>(
919  params, this->GetEncodingParams(), value, depth, level, scFact));
920  }
921 
922  p->Encode();
923  return p;
924  }
