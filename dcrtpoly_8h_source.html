221  virtual DCRTPolyType CloneTowers(uint32_t startTower, uint32_t endTower) const override {
222  vector<NativeInteger> moduli(endTower - startTower + 1);
223  vector<NativeInteger> roots(endTower - startTower + 1);
224 
225  for (uint32_t i = startTower; i <= endTower; i++) {
226  moduli[i - startTower] = this->GetParams()->GetParams()[i]->GetModulus();
227  roots[i - startTower] =
228  this->GetParams()->GetParams()[i]->GetRootOfUnity();
229  }
230 
231  auto params = DCRTPolyImpl::Params(this->GetCyclotomicOrder(), moduli,
232  roots, {}, {}, 0);
233 
234  auto res =
235  DCRTPolyImpl(std::make_shared<typename DCRTPolyImpl::Params>(params),
236  EVALUATION, false);
237 
238  for (uint32_t i = startTower; i <= endTower; i++) {
239  res.SetElementAtIndex(i - startTower, this->GetElementAtIndex(i));
240  }
241 
242  return res;
243  }
