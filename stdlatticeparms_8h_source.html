80 class StdLatticeParm {
81  DistributionType distType;
82  usint ringDim;
83  SecurityLevel minSecLev;
84  usint maxLogQ;
85 
86  // NOTE!!! the declaration below relies upon there being three possible values
87  // for the first index (the distribution type), and three possible values for
88  // the second index (the security level)
89  // The values in the enums, above, meet this criteria
90  // it's also important that the different values are numbered from 0-2
91  // again, the enums above do this
92  // DO NOT change the values of the enums to be anything other than consecutive
93  // numbers starting from 0, or this code will break in strange ways, and you
94  // will suffer MAKE SURE that the number of entries in the DistributionType
95  // enum is == the first index, and MAKE SURE that the number of entries in the
96  // SecurityLevel enum is == the second index
97  static map<usint, StdLatticeParm*> byRing[3][3];
98  static map<usint, StdLatticeParm*> byLogQ[3][3];
99 
100  static vector<StdLatticeParm> StandardLatticeParmSets;
101  static bool initialized;
102 
103  public:
104  StdLatticeParm(DistributionType distType, usint ringDim,
105  SecurityLevel minSecLev, usint maxLogQ)
106  : distType(distType),
107  ringDim(ringDim),
108  minSecLev(minSecLev),
109  maxLogQ(maxLogQ) {}
110 
111  static void initializeLookups() {
112  for (size_t i = 0; i < StandardLatticeParmSets.size(); i++) {
113  StdLatticeParm& s = StandardLatticeParmSets[i];
114  byRing[static_cast<int>(s.distType)][static_cast<int>(s.minSecLev)]
115  [s.ringDim] = &s;
116  byLogQ[static_cast<int>(s.distType)][static_cast<int>(s.minSecLev)]
117  [s.maxLogQ] = &s;
118  }
119  initialized = true;
120  }
121 
122  static usint FindMaxQ(DistributionType distType, SecurityLevel minSecLev,
123  usint ringDim) {
124  int distTypeIdx = static_cast<int>(distType);
125  int minSecLevIdx = static_cast<int>(minSecLev);
126  if (!initialized) initializeLookups();
127  auto it = byRing[distTypeIdx][minSecLevIdx].find(ringDim);
128  if (it == byRing[distTypeIdx][minSecLevIdx].end()) return 0;
129  return it->second->getMaxLogQ();
130  }
131 
132  static usint FindRingDim(DistributionType distType, SecurityLevel minSecLev,
133  usint curLogQ) {
134  if (!initialized) initializeLookups();
135  usint prev = 0;
136 
137  int distTypeIdx = static_cast<int>(distType);
138  int minSecLevIdx = static_cast<int>(minSecLev);
139  for (std::pair<const unsigned int, StdLatticeParm*>& it :
140  byLogQ[distTypeIdx][minSecLevIdx]) {
141  if ((curLogQ <= it.second->getMaxLogQ()) && (curLogQ > prev))
142  return it.second->getRingDim();
143  prev = it.second->getMaxLogQ();
144  }
145  return 65536;
146  }
147 
148  DistributionType getDistType() const { return distType; }
149  usint getRingDim() const { return ringDim; }
150  SecurityLevel getMinSecLev() const { return minSecLev; }
151  usint getMaxLogQ() const { return maxLogQ; }
152 };
