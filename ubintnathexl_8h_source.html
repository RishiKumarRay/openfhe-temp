754  template <typename T = NativeInt>
755  NativeIntegerT Mod(
756  const NativeIntegerT &modulus, const NativeIntegerT &mu,
757  typename std::enable_if<!std::is_same<T, DNativeInt>::value, bool>::type =
758  true) const {
759  typeD tmp1;
760  tmp1.lo = this->m_value;
761  tmp1.hi = 0;
762  DNativeInt tmp(this->m_value);
763 
764  long n = modulus.GetMSB();
765  long alpha = n + 3;
766  long beta = -2;
767 
768  // RShiftD is more efficient than the right-shifting of DNativeInt
769  NativeInt ql = RShiftD(tmp1, n + beta);
770  MultD(ql, mu.m_value, tmp1);
771  DNativeInt q = GetD(tmp1);
772 
773  // we cannot use RShiftD here because alpha - beta > 63
774  // for q larger than 57 bits
775  q >>= alpha - beta;
776  tmp -= q * DNativeInt(modulus.m_value);
777 
778  NativeIntegerT ans;
779  ans.m_value = NativeInt(tmp);
780 
781  // correction at the end
782  if (ans.m_value > modulus.m_value) {
783  ans.m_value -= modulus.m_value;
784  }
785  return ans;
786  }
