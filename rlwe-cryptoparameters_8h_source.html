310  bool operator==(const CryptoParametersBase<Element> &rhs) const {
311  const auto *el = dynamic_cast<const CryptoParametersRLWE<Element> *>(&rhs);
312 
313  if (el == nullptr) return false;
314 
315  return CryptoParametersBase<Element>::operator==(*el) &&
316  this->GetPlaintextModulus() == el->GetPlaintextModulus() &&
317  *this->GetElementParams() == *el->GetElementParams() &&
318  *this->GetEncodingParams() == *el->GetEncodingParams() &&
319  m_distributionParameter == el->GetDistributionParameter() &&
320  m_assuranceMeasure == el->GetAssuranceMeasure() &&
321  m_securityLevel == el->GetSecurityLevel() &&
322  m_noiseScale == el->GetNoiseScale() &&
323  m_relinWindow == el->GetRelinWindow() && m_mode == el->GetMode() &&
324  m_stdLevel == el->GetStdLevel();
