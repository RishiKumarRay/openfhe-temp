33 class ParallelControls {
34  int machineThreads;
35 
36  public:
37  // @Brief CTOR, enables parallel operations as default
38  // Cache the number of machine threads the system reports (can be
39  // overridden by environment variables)
40  // enable on startup by default
41  ParallelControls() {
42 #ifdef PARALLEL
43  machineThreads = omp_get_max_threads();
44  Enable();
45 #else
46  machineThreads = 1;
47 #endif
48  }
49  // @Brief Enable() enables parallel operation
50  void Enable() {
51 #ifdef PARALLEL
52  omp_set_num_threads(machineThreads);
53 #endif
54  }
55  // @Brief Disable() disables parallel operation
56  void Disable() {
57 #ifdef PARALLEL
58  omp_set_num_threads(0);
59 #endif
60  }
61 
62  int GetMachineThreads() const { return machineThreads; }
63 
64  static int GetNumProcs() {
65 #ifdef PARALLEL
66  return omp_get_num_procs();
67 #else
68  return 1;
69 #endif
70  }

72  // @Brief returns current number of threads that are usable
73  // @return int # threads
74  int GetNumThreads() {
75 #ifdef PARALLEL
76  int nthreads = 1;
77  int tid = 1;
78  // Fork a team of threads giving them their own copies of variables
79  // so we can see how many threads we have to work with
80 #pragma omp parallel private(tid)
81  {
82  /* Obtain thread number */
83  tid = omp_get_thread_num();
84 
85  /* Only master thread does this */
86  if (tid == 0) {
87  nthreads = omp_get_num_threads();
88  }
89  }
90  // std::cout << "\nNumber of threads = " << nthreads << std::endl;
91  return nthreads;
92 
93 #else
94  return machineThreads;
95 #endif
96  }
97 
98  // @Brief sets number of threads to use (limited by system value)
99 
100  void SetNumThreads(int nthreads) {
101 #ifdef PARALLEL
102  // set number of thread, but limit it to the system set
103  // number of machine threads...
104  if (nthreads > machineThreads) {
105  nthreads = machineThreads;
106  }
107  omp_set_num_threads(nthreads);
108 #endif
109  }
110 };
