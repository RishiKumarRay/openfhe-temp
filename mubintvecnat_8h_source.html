598  void>::type
599  load(Archive &ar, std::uint32_t const version) {
600  if (version > SerializedVersion()) {
601  PALISADE_THROW(lbcrypto::deserialize_error,
602  "serialized object version " + std::to_string(version) +
603  " is from a later version of the library");
604  }
605  size_t size;
606  ar(size);
607  m_data.resize(size);
608  if (size > 0) {
609  auto *data =
610  reinterpret_cast<IntegerType *>(malloc(size * sizeof(IntegerType)));
611  ar(::cereal::binary_data(data, size * sizeof(IntegerType)));
612  for (size_t i = 0; i < size; i++) {
613  m_data[i] = data[i];
614  }
615  free(data);
616  }
617  ar(m_modulus);
618  }
