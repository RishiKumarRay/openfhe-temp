173  inline MatrixStrassen<Element> ScalarMult(Element const& other) const {
174  MatrixStrassen<Element> result(*this);
175 #pragma omp parallel for
176  for (int32_t col = 0; col < result.cols; ++col) {
177  for (int32_t row = 0; row < result.rows; ++row) {
178  *result.data[row][col] = *result.data[row][col] * other;
179  }
180  }
181  return result;
182  }
