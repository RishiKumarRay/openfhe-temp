773 inline std::string GenerateUniqueKeyID() {
774  const size_t intsInID = 128 / (sizeof(uint32_t) * 8);
775  std::uniform_int_distribution<uint32_t> distribution(
776  0, std::numeric_limits<uint32_t>::max());
777  std::stringstream s;
778  s.fill('0');
779  s << std::hex;
780  for (size_t i = 0; i < intsInID; i++)
781  s << std::setw(8) << distribution(PseudoRandomNumberGenerator::GetPRNG());
782  return s.str();
783 }
