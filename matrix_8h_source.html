213  Matrix<T> GadgetVector(int64_t base = 2) const {
214  Matrix<T> g(allocZero, rows, cols);
215  auto base_matrix = allocZero();
216  size_t k = cols / rows;
217  base_matrix = base;
218  g(0, 0) = 1;
219  for (size_t i = 1; i < k; i++) {
220  g(0, i) = g(0, i - 1) * base_matrix;
221  }
222  for (size_t row = 1; row < rows; row++) {
223  for (size_t i = 0; i < k; i++) {
224  g(row, i + row * k) = g(0, i);
225  }
226  }
227  return g;
228  }

231  std::is_same<T, M2DCRTPoly>::value ||
232  std::is_same<T, M4DCRTPoly>::value ||
233  std::is_same<T, M6DCRTPoly>::value,
234  bool>::type = true>
235  Matrix<T> GadgetVector(int64_t base = 2) const {
236  Matrix<T> g(allocZero, rows, cols);
237  auto base_matrix = allocZero();
238  base_matrix = base;
239  size_t bk = 1;
240 
241  auto params = g(0, 0).GetParams()->GetParams();
242 
243  uint64_t digitCount = (long)ceil(
244  log2(params[0]->GetModulus().ConvertToDouble()) / log2(base));
245 
246  for (size_t k = 0; k < digitCount; k++) {
247  for (size_t i = 0; i < params.size(); i++) {
248  NativePoly temp(params[i]);
249  temp = bk;
250  g(0, k + i * digitCount).SetElementAtIndex(i, std::move(temp));
251  }
252  bk *= base;
253  }
254 
255  size_t kCols = cols / rows;
256  for (size_t row = 1; row < rows; row++) {
257  for (size_t i = 0; i < kCols; i++) {
258  g(row, i + row * kCols) = g(0, i);
259  }
260  }
261  return g;
262  }
