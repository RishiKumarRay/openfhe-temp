218 #define GADGET_FOR_TYPE(T) \
219  template <> \
220  Matrix<T> Matrix<T>::GadgetVector(int64_t base) const { \
221  Matrix<T> g(allocZero, rows, cols); \
222  auto base_matrix = allocZero(); \
223  size_t k = cols / rows; \
224  base_matrix = base; \
225  g(0, 0) = 1; \
226  for (size_t i = 1; i < k; i++) { \
227  g(0, i) = g(0, i - 1) * base_matrix; \
228  } \
229  for (size_t row = 1; row < rows; row++) { \
230  for (size_t i = 0; i < k; i++) { \
231  g(row, i + row * k) = g(0, i); \
232  } \
233  } \
234  return g; \
235  }

237 #define GADGET_FOR_TYPE_DCRT(T) \
238  template <> \
239  Matrix<T> Matrix<T>::GadgetVector(int64_t base) const { \
240  Matrix<T> g(allocZero, rows, cols); \
241  auto base_matrix = allocZero(); \
242  base_matrix = base; \
243  size_t bk = 1; \
244  \
245  auto params = g(0, 0).GetParams()->GetParams(); \
246  \
247  uint64_t digitCount = (long)ceil( \
248  log2(params[0]->GetModulus().ConvertToDouble()) / log2(base)); \
249  \
250  for (size_t k = 0; k < digitCount; k++) { \
251  for (size_t i = 0; i < params.size(); i++) { \
252  NativePoly temp(params[i]); \
253  temp = bk; \
254  g(0, k + i * digitCount).SetElementAtIndex(i, std::move(temp)); \
255  } \
256  bk *= base; \
257  } \
258  \
259  size_t kCols = cols / rows; \
260  for (size_t row = 1; row < rows; row++) { \
261  for (size_t i = 0; i < kCols; i++) { \
262  g(row, i + row * kCols) = g(0, i); \
263  } \
264  } \
265  return g; \
266  }
