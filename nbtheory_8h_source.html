50 inline int clz_u128(unsigned __int128 u) {
51  uint64_t hi = u >> 64;
52  uint64_t lo = u;
53  int retval[3] = {__builtin_clzll(hi), __builtin_clzll(lo) + 64, 128};
54 
55  int idx = !hi + ((!lo) & (!hi));
56  return retval[idx];
57 }

133 /* Function to reverse bits of num */
134 inline usint ReverseBits(usint num, usint msb) {
135  usint msbb = (msb >> 3) + (msb & 0x7 ? 1 : 0);
136 
137  switch (msbb) {
138  case 1:
139  return (reverse_byte((num)&0xff) >> shift_trick[msb & 0x7]);
140 
141  case 2:
142  return (reverse_byte((num)&0xff) << 8 |
143  reverse_byte((num >> 8) & 0xff)) >>
144  shift_trick[msb & 0x7];
145 
146  case 3:
147  return (reverse_byte((num)&0xff) << 16 |
148  reverse_byte((num >> 8) & 0xff) << 8 |
149  reverse_byte((num >> 16) & 0xff)) >>
150  shift_trick[msb & 0x7];
151 
152  case 4:
153  return (reverse_byte((num)&0xff) << 24 |
154  reverse_byte((num >> 8) & 0xff) << 16 |
155  reverse_byte((num >> 16) & 0xff) << 8 |
156  reverse_byte((num >> 24) & 0xff)) >>
157  shift_trick[msb & 0x7];
158  default:
159  return -1;
160  // PALISADE_THROW(math_error, "msbb value not handled:" +
161  // std::to_string(msbb));
162  }

179 template <>
180 inline usint GetMSB(uint64_t x) {
181  if (x == 0) return 0;
182 
183  // hardware instructions for finding MSB are used are used;
184 #if defined(_MSC_VER)
185  // a wrapper for VC++
186  unsigned long msb;
187  _BitScanReverse64(&msb, x);
188  return msb + 1;
189 #else
190  // a wrapper for GCC
191  return 64 -
192  (sizeof(unsigned long) == 8 ? __builtin_clzl(x) : __builtin_clzll(x));
193 #endif
194 }

225  uint32_t k) {
226  auto u_vec = std::make_shared<std::vector<int64_t>>(k);

228  size_t baseDigits = (uint32_t)(std::round(log2(base)));
