92  result_type operator()() {
93  result_type result;
94 
95  if (m_bufferIndex == PRNG_BUFFER_SIZE) m_bufferIndex = 0;
96 
97  // makes a call to the BLAKE2 generator only when the currently buffered
98  // values are all consumed precomputations are done only once for the
99  // current buffer
100  if (m_bufferIndex == 0) Generate();
101 
102  result = m_buffer[m_bufferIndex];
103 
104  m_bufferIndex++;
105 
106  return result;
107  }

123  private:
127  void Generate() {
128  // m_counter is the input to the hash function
129  // m_buffer is the output
130  if (blake2xb(m_buffer.begin(), m_buffer.size() * sizeof(result_type),
131  &m_counter, sizeof(m_counter), m_seed.cbegin(),
132  m_seed.size() * sizeof(result_type)) != 0) {
133  PALISADE_THROW(math_error, "PRNG: blake2xb failed");
134  }
135  m_counter++;
136  return;
137  }
138 
139  // counter used as input to the BLAKE2 hash function
140  // gets incremented after each call
141  uint64_t m_counter = 0;
142 
143  // the seed for the BLAKE2 hash function
144  std::array<result_type, 16> m_seed{};
145 
146  // The vector that stores random samples generated using the hash function
147  std::array<result_type, PRNG_BUFFER_SIZE> m_buffer{};
148 
149  // Index in m_buffer corresponding to the current PRNG sample
150  uint16_t m_bufferIndex = 0;
151 };
