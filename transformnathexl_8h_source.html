40 struct HashPair {
41  template <class T1, class T2>
42  size_t operator()(const std::pair<T1, T2>& p) const {
43  auto hash1 = std::hash<T1>{}(std::get<0>(p));
44  auto hash2 = std::hash<T2>{}(std::get<1>(p));
45  return HashCombine(hash1, hash2);
46  }
47 
48  static size_t HashCombine(size_t lhs, size_t rhs) {
49  lhs ^= rhs + 0x9e3779b9 + (lhs << 6) + (lhs >> 2);
50  return lhs;
51  }
52 };
