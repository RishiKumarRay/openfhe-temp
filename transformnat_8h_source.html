39 struct HashPair {
40  template <class T1, class T2>
41  size_t operator()(const std::pair<T1, T2>& p) const {
42  auto hash1 = std::hash<T1>{}(std::get<0>(p));
43  auto hash2 = std::hash<T2>{}(std::get<1>(p));
44  return HashCombine(hash1, hash2);
45  }
46 
47  static size_t HashCombine(size_t lhs, size_t rhs) {
48  lhs ^= rhs + 0x9e3779b9 + (lhs << 6) + (lhs >> 2);
49  return lhs;
50  }
51 };
