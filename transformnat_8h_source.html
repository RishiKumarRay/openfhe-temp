42 struct HashPair {
43  template <class T1, class T2>
44  size_t operator()(const std::pair<T1, T2>& p) const {
45  auto hash1 = std::hash<T1>{}(std::get<0>(p));
46  auto hash2 = std::hash<T2>{}(std::get<1>(p));
47  return HashCombine(hash1, hash2);
48  }
49 
50  static size_t HashCombine(size_t lhs, size_t rhs) {
51  lhs ^= rhs + 0x9e3779b9 + (lhs << 6) + (lhs >> 2);
52  return lhs;
53  }
54 };
