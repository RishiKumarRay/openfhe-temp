46 inline std::string GenerateUniqueKeyID() {
47  const size_t intsInID = 128 / (sizeof(uint32_t) * 8);
48  std::uniform_int_distribution<uint32_t> distribution(
49  0, std::numeric_limits<uint32_t>::max());
50  std::stringstream s;
51  s.fill('0');
52  s << std::hex;
53  for (size_t i = 0; i < intsInID; i++)
54  s << std::setw(8) << distribution(PseudoRandomNumberGenerator::GetPRNG());
55  return s.str();
56 }
