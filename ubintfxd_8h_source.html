877  // TODO (dsuponit): make ConvertToInt() a template utility function
878  template <typename T = BasicInteger>
879  T ConvertToInt() const {
880  constexpr usint bits = lbcrypto::GetIntegerTypeBitLength<T>();
881  T result = 0;
882  // set num to number of equisized chunks
883  usint num = bits / m_uintBitLength;
884  usint ceilInt = m_nSize - ceilIntByUInt(m_MSB);
885  // copy the values by shift and add
886  for (usint i = 0; i < num && (m_nSize - i - 1) >= ceilInt; i++) {
887  result += ((T)this->m_value[m_nSize - i - 1] << (m_uintBitLength * i));
888  }
889  if (this->m_MSB > bits) {
890  PALISADE_THROW(
891  lbcrypto::math_error,
892  std::string("MSB cannot be bigger than ") +
893  std::to_string(bits));
894  }
895  return result;
896  }

1012  template <typename uint_type_c, usint BITLENGTH_c>
1013  friend std::ostream &operator<<(
1014  std::ostream &os, const BigIntegerFixedT<uint_type_c, BITLENGTH_c> &ptr_obj) {
1015  usint counter;
1016  // initiate to object to be printed
1017  auto print_obj = new BigIntegerFixedT<uint_type_c, BITLENGTH_c>(ptr_obj);
1018  // print_VALUE array stores the decimal value in the array
1019  uschar *print_VALUE = new uschar[ptr_obj.m_numDigitInPrintval];
1020  for (size_t i = 0; i < ptr_obj.m_numDigitInPrintval; i++) {
1021  // reset to zero
1022  *(print_VALUE + i) = 0;
1023  }
1024  // starts the conversion from base r to decimal value
1025  for (size_t i = print_obj->m_MSB; i > 0; i--) {
1026  // print_VALUE = print_VALUE*2
1027  BigIntegerFixedT<uint_type_c, BITLENGTH_c>::double_bitVal(print_VALUE);
1028  // adds the bit value to the print_VALUE
1029  BigIntegerFixedT<uint_type_c, BITLENGTH_c>::add_bitVal(
1030  print_VALUE, print_obj->GetBitAtIndex(i));
1031  }
1032  // find the first occurence of non-zero value in print_VALUE
1033  for (counter = 0; counter < ptr_obj.m_numDigitInPrintval - 1; counter++) {
1034  if (static_cast<int>(print_VALUE[counter]) != 0) {
1035  break;
1036  }
1037  }
1038  // start inserting values into the ostream object
1039  for (; counter < ptr_obj.m_numDigitInPrintval; counter++) {
1040  os << static_cast<int>(print_VALUE[counter]);
1041  }
1042  // deallocate the memory since values are inserted into the ostream object
1043  delete[] print_VALUE;
1044  delete print_obj;
1045  return os;
1046  }
