771  myZZ ModInverse(const myZZ &modulus) const {
772  if (modulus == myZZ(0)) {
773  PALISADE_THROW(lbcrypto::math_error, "zero has no inverse");
774  }
775  myZZ tmp(0);
776  try {
777  tmp = InvMod(*this % modulus, modulus);

785  PALISADE_THROW(lbcrypto::math_error, errmsg.str());
